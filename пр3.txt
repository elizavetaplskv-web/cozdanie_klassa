class StationeryItem:
    def __init__(self, name="", price=0.0, quantity=0, category="", manufacturer=""):
        self._name = name
        self._price = price
        self._quantity = quantity
        self._category = category
        self._manufacturer = manufacturer
    def get_name(self):
        return self._name
    def get_price(self):
        return self._price
    def get_quantity(self):
        return self._quantity
    def get_category(self):
        return self._category
    def get_manufacturer(self):
        return self._manufacturer
    def set_name(self, name):
        if isinstance(name, str) and len(name) > 0:
            self._name = name
        else:
            print("–û—à–∏–±–∫–∞: –Ω–∞–∑–≤–∞–Ω–∏–µ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ –ø—É—Å—Ç–æ–π —Å—Ç—Ä–æ–∫–æ–π")
    def set_price(self, price):
        if isinstance(price, (int, float)) and price >= 0:
            self._price = price
        else:
            print("–û—à–∏–±–∫–∞: —Ü–µ–Ω–∞ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –Ω–µ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º")
    def set_quantity(self, quantity):
        if isinstance(quantity, int) and quantity >= 0:
            self._quantity = quantity
        else:
            print("–û—à–∏–±–∫–∞: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º")
    def set_category(self, category):
        if isinstance(category, str) and len(category) > 0:
            self._category = category
        else:
            print("–û—à–∏–±–∫–∞: –∫–∞—Ç–µ–≥–æ—Ä–∏—è –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –Ω–µ –ø—É—Å—Ç–æ–π —Å—Ç—Ä–æ–∫–æ–π")
    def set_manufacturer(self, manufacturer):
        if isinstance(manufacturer, str) and len(manufacturer) > 0:
            self._manufacturer = manufacturer
        else:
            print("–û—à–∏–±–∫–∞: –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –Ω–µ –ø—É—Å—Ç–æ–π —Å—Ç—Ä–æ–∫–æ–π")
    def calculate_total_value(self):
        return self._price * self._quantity
    def apply_discount(self, discount_percent):
        if 0 <= discount_percent <= 100:
            self._price = self._price * (1 - discount_percent / 100)
            return True
        return False
    def update_stock(self, amount, operation="add"):
        if operation == "add":
            self._quantity += amount
            return f"–î–æ–±–∞–≤–ª–µ–Ω–æ {amount} –µ–¥–∏–Ω–∏—Ü. –ù–æ–≤–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ: {self._quantity}"
        elif operation == "remove":
            if self._quantity >= amount:
                self._quantity -= amount
                return f"–£–¥–∞–ª–µ–Ω–æ {amount} –µ–¥–∏–Ω–∏—Ü. –ù–æ–≤–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ: {self._quantity}"
            else:
                return "–û—à–∏–±–∫–∞: –Ω–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Ç–æ–≤–∞—Ä–∞ –Ω–∞ —Å–∫–ª–∞–¥–µ"
        else:
            return "–ù–µ–∏–∑–≤–µ—Å—Ç–Ω–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è"

import random
from datetime import datetime
class StationeryGame:
    def __init__(self):
        self._inventory = []
        self._balance = 10000
        self._day = 1
        self._customers_served = 0
    def add_product(self, name, price, quantity, category, manufacturer):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Ç–æ–≤–∞—Ä–∞ –≤ –∏–Ω–≤–µ–Ω—Ç–∞—Ä—å"""
        new_item = StationeryItem(name, price, quantity, category, manufacturer)
        self._inventory.append(new_item)
        return f"–¢–æ–≤–∞—Ä '{name}' –¥–æ–±–∞–≤–ª–µ–Ω –≤ –∏–Ω–≤–µ–Ω—Ç–∞—Ä—å"
    def simulate_day(self):
        """–°–∏–º—É–ª—è—Ü–∏—è —Ä–∞–±–æ—á–µ–≥–æ –¥–Ω—è"""
        print(f"\n=== –î–µ–Ω—å {self._day} ===")
        if not self._inventory:
            print("–ù–µ—Ç —Ç–æ–≤–∞—Ä–æ–≤ –¥–ª—è –ø—Ä–æ–¥–∞–∂–∏!")
            return
        daily_profit = 0
        customers_today = random.randint(5, 15)
        for customer in range(customers_today):
            if not self._inventory:
                break
            product = random.choice(self._inventory)
            max_quantity = min(product.get_quantity(), random.randint(1, 10))
            if max_quantity > 0:
                quantity_bought = random.randint(1, max_quantity)
                sale_amount = product.get_price() * quantity_bought
                product.update_stock(quantity_bought, "remove")
                self._balance += sale_amount
                daily_profit += sale_amount
                self._customers_served += 1
                print(f"–ü—Ä–æ–¥–∞–Ω–æ: {quantity_bought} x {product.get_name()} - {sale_amount:.2f} —Ä—É–±.")
        print(f"–ü—Ä–∏–±—ã–ª—å –∑–∞ –¥–µ–Ω—å: {daily_profit:.2f} —Ä—É–±.")
        print(f"–û–±—Å–ª—É–∂–µ–Ω–æ –∫–ª–∏–µ–Ω—Ç–æ–≤: {customers_today}")
        print(f"–û–±—â–∏–π –±–∞–ª–∞–Ω—Å: {self._balance:.2f} —Ä—É–±.")
        self._day += 1
    def show_inventory(self):
        print("\n=== –¢–µ–∫—É—â–∏–π –∏–Ω–≤–µ–Ω—Ç–∞—Ä—å ===")
        if not self._inventory:
            print("–ò–Ω–≤–µ–Ω—Ç–∞—Ä—å –ø—É—Å—Ç")
            return
        total_value = 0
        for i, item in enumerate(self._inventory, 1):
            item_value = item.calculate_total_value()
            total_value += item_value
            print(f"{i}. {item.get_name()} - {item.get_quantity()} —à—Ç. - {item_value:.2f} —Ä—É–±.")
        print(f"–û–±—â–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å –∏–Ω–≤–µ–Ω—Ç–∞—Ä—è: {total_value:.2f} —Ä—É–±.")
    def restock_product(self, product_index, quantity):
        if 0 <= product_index < len(self._inventory):
            product = self._inventory[product_index]
            cost = product.get_price() * quantity * 0.7 
            if self._balance >= cost:
                self._balance -= cost
                product.update_stock(quantity, "add")
                return f"–ó–∞–∫—É–ø–ª–µ–Ω–æ {quantity} –µ–¥–∏–Ω–∏—Ü {product.get_name()}. –°—Ç–æ–∏–º–æ—Å—Ç—å: {cost:.2f} —Ä—É–±."
            else:
                return "–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Å—Ä–µ–¥—Å—Ç–≤ –¥–ª—è –∑–∞–∫—É–ø–∫–∏"
        else:
            return "–¢–æ–≤–∞—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω"
    def get_game_stats(self):
        return {
            'day': self._day,
            'balance': self._balance,
            'customers_served': self._customers_served,
            'products_count': len(self._inventory),
            'total_inventory_value': sum(item.calculate_total_value() for item in self._inventory)
        }
